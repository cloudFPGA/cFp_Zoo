71 # proc my_puts { text {fg -1} }
72 # A procedure that adds a specific prompt and that changes the color of the
73 # text to display by calling the 'tput utility of Linux.
74 # :param text the text to display as a string.
75 # :param fg the foreground color to set (0=Black, 1=Red, 2=Green, 3=Yellow,
76 # 4=Blue, 5=Magenta, 6=Cyan, 7=White).
77 # :return nothing
78 #
79 # See https://linux.101hacks.com/ps1-examples/prompt-color-using-tput/
80 #-------------------------------------------------------------------------------
81 proc my_puts { text {fg -1} } {
82  set prompt "<cloudFPGA> "
83  if { ${fg} != -1 && ${fg} >= 0 && ${fg} <= 7 } {
84  set color ${fg}
85  } else {
86  set color ${::BLUE}
87  }
88  if { ${::OS} eq "WINDOWS" } {
89  set coloredText "${prompt}${text}"
90  } else {
91  set coloredText [exec tput setaf ${color}]${prompt}${text}[exec tput sgr0]
92  }
93  puts ${coloredText}
94 }

127 # proc my_dbg_trace { text dbgLvl }
128 # A procedure to display a debug message colored as a function of the global
129 # debug level set in this file and the debug level requested by the caller.
130 # :param text the text to display as a string.
131 # :param dbgLvl the current debug level used by the caller.
132 # :return nothing
133 #-------------------------------------------------------------------------------
134 proc my_dbg_trace { text dbgLvl } {
135  if { ${::gDbgLvl} == 0 } {
136  return
137  } else {
138  switch ${dbgLvl} {
139  1 { if { ${::gDbgLvl} >= 1 } { my_puts "<D1> ${text}" ${::CYAN}}}
140  2 { if { ${::gDbgLvl} >= 2 } { my_puts "<D2> ${text}" ${::GREEN}}}
141  3 { if { ${::gDbgLvl} >= 3 } { my_puts "<D3> ${text}" ${::YELLOW}}}
142  default { my_puts "<UNKNOWN DEBUG LEVEL> ${text}"}
143  }
144  }
145 }
