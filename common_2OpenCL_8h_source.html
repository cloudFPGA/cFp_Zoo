827  const auto start = std::chrono::high_resolution_clock::now();
828  auto event = ExecuteTaskFork();
829  event.wait();
830  const auto end = std::chrono::high_resolution_clock::now();
831  const double elapsedChrono =
832  1e-9 * std::chrono::duration_cast<std::chrono::nanoseconds>(end - start)
833  .count();

836  event.getProfilingInfo(CL_PROFILING_COMMAND_START, &timeStart);
837  event.getProfilingInfo(CL_PROFILING_COMMAND_END, &timeEnd);
838  const double elapsedSDAccel = 1e-9 * (timeEnd - timeStart);
839  return {elapsedSDAccel, elapsedChrono};
840 #else
841  return {elapsedChrono, elapsedChrono};
842 #endif
843  }

899 Program Context::MakeProgram(std::string const &path) {
900 #ifndef HLSLIB_SIMULATE_OPENCL
901  std::ifstream input(path, std::ios::in | std::ios::binary | std::ios::ate);
902  if (!input.is_open()) {
903  std::stringstream ss;
904  ss << "Failed to open program file \"" << path << "\".";
905  ThrowConfigurationError(ss.str());
906  }
907 
908  // Determine size of file in bytes
909  input.seekg(0, input.end);
910  const auto fileSize = input.tellg();
911  input.seekg(0, input.beg);
912 
913  // Load content of binary file into a character vector (required by the
914  // OpenCL C++ bindings).
915  // The constructor of cl::Program accepts a vector of binaries, so stick the
916  // binary into a single-element outer vector.
917  std::vector<std::vector<unsigned char>> binaryContent;
918  binaryContent.emplace_back(fileSize);
919  // Since this is just binary data the reinterpret_cast *should* be safe
920  input.read(reinterpret_cast<char *>(&binaryContent[0][0]), fileSize);
921 #ifndef HLSLIB_LEGACY_OPENCL
922  const auto binary = binaryContent;
923 #else
924  // In the legacy API, each binary is a void* to the data and an integer
925  // containing its size in bytes
926  typename cl::Program::Binaries binary;
927  binary.emplace_back(&binaryContent[0][0], fileSize);
928 #endif
929 
930  if (fileSize == 0 || binaryContent[0].size() != fileSize) {
931  std::stringstream ss;
932  ss << "Failed to read binary file \"" << path << "\".";
933  ThrowConfigurationError(ss.str());
934  }
935 
936  cl_int errorCode;
937 
938  // Create OpenCL program
939  std::vector<int> binaryStatus(1);
940  std::vector<cl::Device> devices;
941  devices.emplace_back(device_);
942  auto program = std::make_shared<std::pair<std::string, cl::Program>>(
943  path, cl::Program(context_, devices, binary, &binaryStatus, &errorCode));
944  if (binaryStatus[0] != CL_SUCCESS || errorCode != CL_SUCCESS) {
945  std::stringstream ss;
946  ss << "Failed to create OpenCL program from binary file \"" << path
947  << "\":";
948  if (binaryStatus[0] != CL_SUCCESS) {
949  ss << " binary status: " << binaryStatus[0] << ".";
950  }
951  if (errorCode != CL_SUCCESS) {
952  ss << " error code: " << errorCode << ".";
953  }
954  ThrowConfigurationError(ss.str());
955  }
956 
957  // Build OpenCL program
958  {
959  std::lock_guard<std::mutex> lock(reprogramMutex_);
960  errorCode = program->second.build(devices, nullptr, nullptr, nullptr);
961  }
962  if (errorCode != CL_SUCCESS) {
963  std::stringstream ss;
964  ss << "Failed to build OpenCL program from binary file \"" << path << "\".";
965  ThrowConfigurationError(ss.str());
966  }
967 #else
968  decltype(loadedProgram_) program = nullptr;
969 #endif
970 
971  loadedProgram_ = program;
972  return Program(*this, loadedProgram_);
973 }
