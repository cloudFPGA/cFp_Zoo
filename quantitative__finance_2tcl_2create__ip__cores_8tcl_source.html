56 # my_clean_ip_dir ${ipModName}
57 # A procedure that deletes the previous IP directory (if exists).
58 # :param ipModName the name of the IP module to clean.
59 # :return 0 if OK
60 #-------------------------------------------------------------------------------
61 proc my_clean_ip_dir { ipModName } {
62 
63  ## STEP-1: Open the project if it is not yet open.
64  if { [ catch { current_project} rc] } {
65  my_warn "Project \'${rc}\' is not opened. Openning now!"
66  open_project ${::ipXprFile}
67  }
68 
69  ## STEP-2: Check if a FileSet is already in use in the project.
70  set filesets [ get_filesets]
71  foreach fileset $filesets {
72  if [ string match "${::ipDir}/${ipModName}/${ipModName}.xci" ${fileset}] {
73  my_dbg_trace "An IP with name \'${ipModName}\' already exists and will be removed from the project \'${::ipXprName} !" ${::dbgLvl_1}
74  remove_files -fileset ${ipModName} ${fileset}
75  break
76  }
77  }
78 
79  ## STEP-3: Check for possible dangling files (may happen after IP flow got broken)
80  set files [get_files]
81  foreach file $files {
82  if [ string match "${::ipDir}/${ipModName}/${ipModName}.xci" ${file}] {
83  my_dbg_trace "A dangling IP file \'${ipModName}\' already exists and will be removed from the project \'${::ipXprName} !" ${::dbgLvl_1}
84  remove_files ${ipModName} ${file}
85  break
86  }
87  }
88 
89  ## STEP-4: Delete related previous files and directories from the disk
90  file delete -force ${::ipDir}/ip_user_files/ip/${ipModName}
91  file delete -force ${::ipDir}/ip_user_files/sim_scripts/${ipModName}
92 
93  file delete -force ${::ipDir}/${ipModName}
94 
95  return ${::OK}
96 }

100 # my_create_ip ${ipModName ipDir ipVendor ipLibrary ipName ipVersion ipCfgList}
101 # A procedure that automates the creation and the customizationof of an IP.
102 # :param ipModName the name of the IP module to create (as a string).
103 # :param ipDir the directory path for remote IP to be created and
104 # managed outside of the project (as a string).
105 # :param ipVendor the IP vendor name (as a string).
106 # :param ipLibrary the IP library name (as a string).
107 # :param ipName the IP name (as a string").
108 # :param ipVersion the IP version (as a string).
109 # :param ipCfgList a list of name/value pairs of properties to set.
110 # :return 0 if OK
111 #-------------------------------------------------------------------------------
112 proc my_customize_ip {ipModName ipDir ipVendor ipLibrary ipName ipVersion ipCfgList} {
113 
114  if { ${::gTargetIpCore} ne ${ipModName} && ${::gTargetIpCore} ne "all" } {
115  # Skip the creation and customization this IP module
116  my_dbg_trace "Skipping Module ${ipModName}" ${::dbgLvl_1}
117  return ${::OK}
118  } else {
119  set ::nrGenIPs [ expr { ${::nrGenIPs} + 1 }]
120  }
121 
122  set title "## Creating IP Module: ${ipModName} "
123  while { [ string length $title] < 80 } {
124  append title "#"
125  }
126  my_puts "${title}"
127  my_puts " (in remote directory: ${ipDir})"
128  my_puts " ipName = ${ipName}"
129  my_puts " ipVendor = ${ipVendor}"
130  my_puts " ipLibrary = ${ipLibrary}"
131  my_puts " ipVersion = ${ipVersion}"
132  my_puts " ipCfgList = ${ipCfgList}"
133 
134  set rc [ my_clean_ip_dir "${ipModName}"]
135 
136  my_dbg_trace "Done with \'my_clean_ip\' (RC=${rc})." ${::dbgLvl_1}
137 
138  # Note-1: A typical 'create_ip" command looks like the following:
139  # "create_ip -name axis_register_slice -vendor xilinx.com -library ip \
140  # -version 1.1 -module_name ${ipModName} -dir ${ipDir}"
141  #
142  # Note-2: Intercepting errors from 'create_ip' with 'catch' does not work for me.
143  # Therefore, I decided to use the '-quiet' option which execute the command quietly,
144  # returning no messages from the command and always returning 'TCL_OK' regardless of
145  # any errors encountered during execution.
146  if { [ catch { create_ip -name ${ipName} -vendor ${ipVendor} -library ${ipLibrary} \
147  -version ${ipVersion} -module_name ${ipModName} -dir ${ipDir} -quiet} errMsg] } {
148  puts "ERROR_MSG = " ${errMsg}
149  my_err_puts "## The TCL command \'create_ip\' failed (Error message = ${errMsg}"
150  return ${::KO}
151  }
152 
153  my_dbg_trace "Done with \'create_ip\' (errMsg=${errMsg})." ${::dbgLvl_1}
154 
155  if { ! [ string match ${ipModName} [ get_ips ${ipModName}]] } {
156  # The returned list does not match expected value. IP was not created.
157  my_err_puts "## Failed to create IP \'${ipModName}\'."
158  my_puts ""
159  return ${::KO}
160  }
161 
162  # Note: A typical 'set_property' command looks like the following:
163  # "set_property -dict [ list CONFIG.TDATA_NUM_BYTES {8} \
164  # CONFIG.HAS_TKEEP {1} \
165  # CONFIG.HAS_TLAST {1} ] [ get_ips ${ipModName} ]
166  if { [llength ${ipCfgList}] != 0 } {
167  set_property -dict ${ipCfgList} [ get_ips ${ipModName}]
168  my_dbg_trace "Done with \'set_property\'." ${::dbgLvl_1}
169  } else {
170  my_dbg_trace "There is no \'set_property\' to be done." ${::dbgLvl_2}
171  }
172 
173  generate_target {instantiation_template} \
174  [ get_files ${ipDir}/${ipModName}/${ipModName}.xci]
175  my_dbg_trace "Done with \'generate_target\'." ${::dbgLvl_1}
176 
177  update_compile_order -fileset sources_1
178 
179  generate_target all [ get_files ${ipDir}/${ipModName}/${ipModName}.xci]
180  my_dbg_trace "Done with \'generate_target all\'." ${::dbgLvl_1}
181 
182  catch { config_ip_cache -export [ get_ips -all ${ipModName}]}
183 
184  export_ip_user_files -of_objects \
185  [ get_files ${ipDir}/${ipModName}/${ipModName}.xci] -no_script -sync -force -quiet
186  my_dbg_trace "Done with \'export_ip_user_files\'." ${::dbgLvl_1}
187 
188  create_ip_run [ get_files -of_objects \
189  [ get_fileset sources_1] ${ipDir}/${ipModName}/${ipModName}.xci]
190  my_dbg_trace "Done with \'create_ip_run\'." ${::dbgLvl_1}
191 
192  puts "## Done with IP \'${ipModName}\' creation and customization. \n"
193  return ${::OK}
194 
195 }

215 # Parsing of the Command Line
216 # Note: All the strings after the '-tclargs' option are considered as TCL
217 # arguments to this script and are passed on to TCL 'argc' and 'argv'.
218 #-------------------------------------------------------------------------------

220  my_dbg_trace "There are [ llength $argv] TCL arguments to this script." ${dbgLvl_1}
221  set i 0
222  foreach arg $argv {
223  my_dbg_trace " argv\[$i\] = $arg " ${dbgLvl_2}
224  set i [ expr { ${i} + 1 }]
225  }
226  # Step-1: Processing of '$argv' using the 'cmdline' package
227  set options {
228  { ipModuleName.arg "all" "The module name of the IP to create." }
229  { ipList "Display the list of IP modules names." }
230  }
231  set usage "\nUSAGE: vivado -mode batch -source ${argv0} -notrace -tclargs \[OPTIONS] \nOPTIONS:"
232 
233  array set kvList [ cmdline::getoptions argv ${options} ${usage}]
234 
235  # Process the arguments
236  foreach { key value } [ array get kvList] {
237  my_dbg_trace "KEY = ${key} | VALUE = ${value}" ${dbgLvl_2}
238  if { ${key} eq "ipModuleName" && ${value} ne "all" } {
239  set gTargetIpCore ${value}
240  my_dbg_trace "Setting gTargetIpCore to \'${gTargetIpCore}\' " ${dbgLvl_1}
241  break;
242  }
243  if { ${key} eq "ipList" } {
244  my_puts "IP MODULE NAMES THAT CAN GENERATED BY THIS SCRIPT:"
245  set thisScript [ open ${argv0}]
246  set lines [ split [ read $thisScript] \n]
247  foreach line ${lines} {
248  if { [ string match "set*ipModName*" ${line}] } {
249  if { [ llength [ split ${line}]] == 3 } {
250  if { [ lindex [ split ${line}] 0] eq "set" && \
251  [ lindex [ split ${line}] 1] eq "ipModName" } {
252  set modName [ lindex [ split ${line}] 2]
253  puts "\t${modName}"
254  }
255  }
256  }
257  }
258  close ${thisScript}
259  return ${::OK}
260  }
261  }
262 }
