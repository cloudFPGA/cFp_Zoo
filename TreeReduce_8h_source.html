16 template <typename T, class Operator, int width>
17 struct TreeReduceImplementation {
18  template <typename RandomAccessType>
19  static T f(RandomAccessType const &arr) {
20  #pragma HLS INLINE
21  static constexpr int halfWidth = width / 2;
22  static constexpr int reducedSize = halfWidth + width % 2;
23  T reduced[reducedSize];
24  #pragma HLS ARRAY_PARTITION variable=reduced complete
25  Reduction:
26  for (int i = 0; i < halfWidth; ++i) {
27  #pragma HLS UNROLL
28  reduced[i] = Operator::Apply(arr[i * 2], arr[i * 2 + 1]);
29  }
30  if (halfWidth != reducedSize) {
31  reduced[reducedSize - 1] = arr[width - 1];
32  }
33  return TreeReduceImplementation<T, Operator, reducedSize>::f(reduced);
34  }
